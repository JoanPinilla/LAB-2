Global Integer h

Function main

Motor On
Power High

Accel 50, 50
Speed 50

h = 550

Home
Do
	Wait 0.5
	Go Origen
	Wait 0.5
	Move Ejex
	Wait 0.5
	Go Ejey
	Wait 0.5
	Call paletizado_z
	Call paletizado_s
Loop

Fend

Function paletizado_z
	Integer i
	Pallet 1, Origen, Ejey, Ejex, 3, 3
	For i = 1 To 9
		'Go Pallet(1, i) ' :Z(h)
		Jump Pallet(1, i)
		'Go Pallet(1, i) ' :Z(h)
	Next
	i = 0
Fend

Function paletizado_s
	Integer i, k
	Pallet 2, Origen, Ejey, Ejex, 3, 3
	For i = 1 To 9
		If i >= 4 And i <= 6 Then
		Print "hell yeah"
			For k = 6 To 4 Step -1
				Go Pallet(1, k) :Z(h)
				Go Pallet(1, k)
				Go Pallet(1, k) :Z(h)
			Next
		EndIf
		Go Pallet(1, i) :Z(h)
		Go Pallet(1, i)
		Go Pallet(1, i) :Z(h)
	Next
	i = 0
Fend

Function paletizado_externo
	Integer i, k
	Pallet 2, Origen, Ejey, Ejex, 3, 3
	For i = 1 To 9
		If i >= 4 And i <= 6 Then
		Print "hell yeah"
			For k = 6 To 4 Step -1
				Go Pallet(1, k) :Z(h)
			Next
		EndIf
		Go Pallet(1, i) :Z(h)
	Next
	i = 0
Fend
